const Particles = {
  canvas: document.getElementById('particles-canvas'),
  ctx: null,
  particles: [],
  init() {
    this.ctx = this.canvas.getContext('2d');
    this.resize();
    window.addEventListener('resize', () => this.resize());
    this.createParticles(50);
    this.animate();
  },
  resize() {
    this.canvas.width = window.innerWidth;
    this.canvas.height = window.innerHeight;
  },
  createParticles(count) {
    for (let i = 0; i < count; i++) {
      this.particles.push({
        x: Math.random() * this.canvas.width,
        y: this.canvas.height + Math.random() * 100,
        size: Math.random() * 2 + 1,
        speed: Math.random() * 0.5 + 0.2,
        color: ['#c8ff00', '#00e5ff', '#ffd44d'][Math.floor(Math.random() * 3)],
        opacity: Math.random() * 0.5 + 0.1
      });
    }
  },
  animate() {
    this.ctx.clearRect(0, 0, this.canvas.width, this.canvas.height);
    this.particles.forEach(p => {
      p.y -= p.speed;
      if (p.y < -10) {
        p.y = this.canvas.height + 10;
        p.x = Math.random() * this.canvas.width;
      }
      this.ctx.beginPath();
      this.ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
      this.ctx.fillStyle = p.color + Math.floor(p.opacity * 255).toString(16).padStart(2, '0');
      this.ctx.fill();
    });
    requestAnimationFrame(() => this.animate());
  }
};

// ===== SCALES MODAL =====
function openScalesModal(productId) {
  const product = state.catalog.find(p => p.id === productId);
  if (!product) return;

  state.currentProduct = product;
  state.selectedScales = {};

  els.modalProductName.textContent = product.name;
  els.scalesList.innerHTML = '';

  const scaleLabels = {
    'detal': 'Detal (1-9)',
    'x10': 'Mayor ×10',
    'x50': 'Mayor ×50',
    'x100': 'Mayor ×100',
    'x500': 'Mayor ×500',
    'x1000': 'Mayor ×1000'
  };

  const scales = Object.entries(product.scales)
    .filter(([key]) => scaleLabels[key])
    .map(([key, price]) => ({ key, label: scaleLabels[key], price }));

  if (scales.length === 0) {
    els.scalesList.innerHTML = '<p style="color:var(--muted2);text-align:center;padding:40px">Sin escalas disponibles</p>';
  } else {
    scales.forEach(scale => {
      const row = document.createElement('div');
      row.className = 'scale-row';
      row.innerHTML = `
        <span class="scale-label">${scale.label}</span>
        <span class="scale-price">$${scale.price.toLocaleString('es-CO')}</span>
        <div class="scale-qty">
          <button onclick="updateScaleQty('${scale.key}', -1)">−</button>
          <input type="number" value="0" min="0" data-scale="${scale.key}" onchange="updateModalTotal()">
          <button onclick="updateScaleQty('${scale.key}', 1)">+</button>
        </div>
      `;
      els.scalesList.appendChild(row);
    });
  }

  updateModalTotal();
  els.scalesModal.classList.add('active');
}

window.updateScaleQty = function(scale, delta) {
  const input = document.querySelector(`input[data-scale="${scale}"]`);
  let val = parseInt(input.value) || 0;
  val += delta;
  if (val < 0) val = 0;
  input.value = val;
  updateModalTotal();
};

function updateModalTotal() {
  if (!state.currentProduct) return;

  let total = 0;
  els.scalesList.querySelectorAll('.scale-row').forEach(row => {
    const scale = row.querySelector('input').dataset.scale;
    const qty = parseInt(row.querySelector('input').value) || 0;
    const price = state.currentProduct.scales[scale];
    if (price) total += qty * price;
  });

  els.modalTotal.textContent = `$${total.toLocaleString('es-CO')}`;
}

els.modalAddCart.addEventListener('click', () => {
  if (!state.currentProduct) return;

  let added = false;
  let totalQty = 0;
  
  els.scalesList.querySelectorAll('.scale-row').forEach(row => {
    const scale = row.querySelector('input').dataset.scale;
    const qty = parseInt(row.querySelector('input').value) || 0;
    if (qty > 0 && state.currentProduct.scales[scale]) {
      const price = state.currentProduct.scales[scale];
      addToCart(state.currentProduct.id, scale, qty, price);
      added = true;
      totalQty += qty;
    }
  });

  if (added) {
    els.scalesModal.classList.remove('active');
    updateCartUI();
    updateRecommendations();
    els.cartDrawer.classList.add('active');
    showToast(`¡${totalQty} unidad(es) agregada(s) al carrito!`, 'success');
  } else {
    showToast('Selecciona una cantidad', 'warning');
  }
});

els.modalClose.addEventListener('click', () => {
  els.scalesModal.classList.remove('active');
});

// ===== CART =====
function addToCart(productId, scale, quantity, unitPrice) {
  const existing = state.cart.find(item => item.productId === productId && item.scale === scale);
  if (existing) {
    existing.quantity += quantity;
  } else {
    state.cart.push({ productId, scale, quantity, unitPrice });
  }
  saveCart();
}

function removeFromCart(index) {
  state.cart.splice(index, 1);
  saveCart();
  updateCartUI();
  updateRecommendations();
  showToast('Producto eliminado del carrito', 'info');
}

function updateCartItemQty(index, delta) {
  if (state.cart[index]) {
    state.cart[index].quantity += delta;
    if (state.cart[index].quantity <= 0) removeFromCart(index);
    else {
      saveCart();
      updateCartUI();
    }
  }
}

function saveCart() {
  localStorage.setItem('vs_cart', JSON.stringify(state.cart));
}

function updateCartUI() {
  const count = state.cart.reduce((sum, item) => sum + item.quantity, 0);
  els.cartCount.textContent = count;
  els.cartCount.style.display = count > 0 ? 'flex' : 'none';

  els.cartItems.innerHTML = '';
  if (state.cart.length === 0) {
    els.cartItems.innerHTML = '<p style="text-align:center;color:var(--muted2);padding:60px 20px;font-size:0.9rem">Tu carrito está vacío<br><small>Agrega productos para comenzar</small></p>';
  } else {
    state.cart.forEach((item, index) => {
      const product = state.catalog.find(p => p.id === item.productId);
      if (!product) return;

      const el = document.createElement('div');
      el.className = 'cart-item';
      el.innerHTML = `
        <div class="cart-item-info">
          <div class="cart-item-name">${product.name}</div>
          <div class="cart-item-scale">${item.scale.replace('x', '×')}</div>
          <div class="cart-item-controls">
            <div class="cart-item-qty">
              <button onclick="updateCartItemQty(${index}, -1)">−</button>
              <span>${item.quantity}</span>
              <button onclick="updateCartItemQty(${index}, 1)">+</button>
            </div>
          </div>
        </div>
        <div class="cart-item-price">$${(item.unitPrice * item.quantity).toLocaleString('es-CO')}</div>
        <button class="cart-item-delete" onclick="removeFromCart(${index})">×</button>
      `;
      els.cartItems.appendChild(el);
    });
  }

  const total = state.cart.reduce((sum, item) => sum + (item.unitPrice * item.quantity), 0);
  els.cartTotal.textContent = `$${total.toLocaleString('es-CO')}`;
  els.checkoutBtn.disabled = state.cart.length === 0;
}

// ===== RECOMMENDATIONS CAROUSEL =====
function updateRecommendations() {
  if (state.cart.length === 0) {
    els.recommendations.style.display = 'none';
    return;
  }
  els.recommendations.style.display = 'block';

  const cartCategories = [...new Set(state.cart.map(item => {
    const p = state.catalog.find(c => c.id === item.productId);
    return p?.category;
  }).filter(Boolean))];

  const recs = state.catalog.filter(p => 
    !state.cart.some(c => c.productId === p.id) &&
    (cartCategories.includes(p.category) || Math.random() > 0.5)
  ).slice(0, 6);

  els.recCarousel.innerHTML = '';
  if (recs.length === 0) {
    els.recCarousel.innerHTML = '<p style="font-size:0.75rem;color:var(--muted2);padding:20px">Explora más productos</p>';
    return;
  }

  recs.forEach(product => {
    const el = document.createElement('div');
    el.className = 'rec-item';
    el.innerHTML = `
      <div class="rec-item-img">
        ${product.img.startsWith('http') ? `<img src="${product.img}" alt="${product.name}">` : `<span style="font-size:2rem">${product.img || '📦'}</span>`}
      </div>
      <div class="rec-item-name">${product.name}</div>
      <div class="rec-item-price">$${product.price.toLocaleString('es-CO')}</div>
    `;
    el.addEventListener('click', () => {
      openScalesModal(product.id);
      els.cartDrawer.classList.remove('active');
    });
    els.recCarousel.appendChild(el);
  });
}

els.productSave.addEventListener('click', () => {
  const id = parseInt(els.editProductId.value);
  const name = els.productName.value.trim();
  const category = els.productCategory.value;
  const desc = els.productDesc.value.trim();
  const img = els.productImg.value.trim();
  const badge = els.productBadge.value;

  const scales = {};
  document.querySelectorAll('.scale-admin-row').forEach(row => {
    const key = row.querySelector('.scale-key').value.trim();
    const price = parseInt(row.querySelector('.scale-price-input').value);
    if (key && price) scales[key] = price;
  });

  if (!name || !category || Object.keys(scales).length === 0) {
    showToast('Completa los campos obligatorios', 'error');
    return;
  }

  const price = Math.min(...Object.values(scales));

  if (id) {
    const product = state.catalog.find(p => p.id === id);
    if (product) {
      product.name = name;
      product.category = category;
      product.desc = desc;
      product.img = img || '📦';
      product.badge = badge || null;
      product.scales = scales;
      product.price = price;
      showToast('Producto actualizado', 'success');
    }
  } else {
    const newProduct = {
      id: Date.now(),
      name,
      category,
      price,
      scales,
      desc,
      img: img || '📦',
      badge: badge || null
    };
    state.catalog.push(newProduct);
    showToast('Producto creado', 'success');
  }

  saveCatalog();
  renderCatalog(document.querySelector('.cat-tab.active')?.dataset.cat || 'all');
  closeAdminProductForm();
});

    const item = document.createElement('div');
    item.className = 'admin-category-item';
    item.style.cssText = 'display:flex;align-items:center;gap:8px;padding:8px 12px;background:rgba(255,255,255,.03);border:1px solid var(--border);border-radius:6px;font-size:0.55rem;font-family:JetBrains Mono,monospace';
    item.innerHTML = `
      <span>${cat.name}</span>
      <button onclick="deleteCategory('${cat.id}')" style="background:none;border:none;color:var(--muted2);cursor:pointer;padding:2px">×</button>
    `;
    els.adminCategoryList.appendChild(item);
  });
}

window.deleteCategory = function(categoryId) {
  if (confirm('¿Eliminar esta categoría? Los productos no se eliminarán.')) {
    state.categories = state.categories.filter(c => c.id !== categoryId);
    saveCatalog();
    renderCategories();
    renderAdminCategories();
    updateAdminCategoryDropdown();
    showToast('Categoría eliminada', 'success');
  }
};

els.addCategoryBtn.addEventListener('click', () => {
  const name = els.newCategoryName.value.trim();
  const id = els.newCategoryId.value.trim().toLowerCase().replace(/[^a-z0-9-]/g, '-');

  if (!name || !id) {
    showToast('Ingresa nombre e ID de categoría', 'error');
    return;
  }

  if (state.categories.some(c => c.id === id)) {
    showToast('El ID de categoría ya existe', 'error');
    return;
  }

  state.categories.push({ id, name });
  saveCatalog();
  renderCategories();
  renderAdminCategories();
  updateAdminCategoryDropdown();

  els.newCategoryName.value = '';
  els.newCategoryId.value = '';
  showToast('Categoría creada', 'success');
});

// ===== ADMIN EVENT LISTENERS =====
function setupAdminListeners() {
  els.adminAddProduct.addEventListener('click', () => {
    closeAdminProductForm();
    els.editProductId.value = '';
    els.productForm.classList.remove('hidden');
  });

  els.adminAddCategory.addEventListener('click', () => {
    els.newCategoryName.focus();
  });

  els.adminExport.addEventListener('click', () => {
    const data = {
      catalog: state.catalog,
      categories: state.categories,
      exportedAt: new Date().toISOString()
    };
    const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `vapingstreet-export-${Date.now()}.json`;
    a.click();
    URL.revokeObjectURL(url);
    showToast('Datos exportados', 'success');
  });
}

// ===== EVENT LISTENERS =====
function setupEventListeners() {
  els.categories.addEventListener('click', (e) => {
    if (e.target.classList.contains('cat-tab') && !e.target.classList.contains('admin-add')) {
      document.querySelectorAll('.cat-tab').forEach(t => t.classList.remove('active'));
      e.target.classList.add('active');
      renderCatalog(e.target.dataset.cat);
    }
  });

  els.cartToggle.addEventListener('click', () => {
    els.cartDrawer.classList.add('active');
  });

  els.cartClose.addEventListener('click', () => {
    els.cartDrawer.classList.remove('active');
  });

  [els.scalesModal, els.checkoutModal, els.adminModal, els.productDetailModal].forEach(modal => {
    modal.addEventListener('click', (e) => {
      if (e.target === modal) modal.classList.remove('active');
    });
  });
}

// ===== TOAST NOTIFICATIONS =====
function showToast(message, type = 'info') {
  const container = document.getElementById('toast-container');
  const toast = document.createElement('div');
  toast.className = `toast ${type}`;
  toast.textContent = message;
  container.appendChild(toast);

  setTimeout(() => {
    toast.style.opacity = '0';
    toast.style.transform = 'translateY(-10px)';
    setTimeout(() => toast.remove(), 300);
  }, 3000);
}

// ===== KEYBOARD SHORTCUTS =====
document.addEventListener('keydown', (e) => {
  if (e.key === 'Escape') {
    [els.scalesModal, els.checkoutModal, els.adminModal, els.cartDrawer, els.productDetailModal, els.lightbox].forEach(el => {
      el.classList.remove('active');
    });
    closeAdminProductForm();
  }
  if (e.key === 'c' && e.ctrlKey) {
    e.preventDefault();
    els.cartDrawer.classList.toggle('active');
  }
});
